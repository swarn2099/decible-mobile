**Step 1 — Create `src/hooks/useDiscoverArtist.ts`:**

This hook calls the existing `/mobile/discover` endpoint (or if that doesn't exist, check for `/api/discover` — read the research to confirm). The discover action creates a "discovered" relationship between the current user and an existing performer.

```typescript
import { useMutation, useQueryClient } from "@tanstack/react-query";
import * as Haptics from "expo-haptics";
import { apiCall } from "@/lib/api";

type DiscoverInput = { performerId: string };
type DiscoverResult = { success: boolean; performer: { id: string; name: string; slug: string } };

export function useDiscoverArtist() {
  const queryClient = useQueryClient();
  return useMutation<DiscoverResult, Error, DiscoverInput>({
    mutationFn: async ({ performerId }) => {
      return apiCall<DiscoverResult>("/mobile/discover", {
        method: "POST",
        body: JSON.stringify({ performerId }),
      });
    },
    onSuccess: () => {
      Haptics.impactAsync(Haptics.ImpactFeedbackStyle.Medium);
      queryClient.invalidateQueries({ queryKey: ["passport"] });
      queryClient.invalidateQueries({ queryKey: ["fanBadges"] });
    },
  });
}
```

**Note:** If `/mobile/discover` doesn't exist yet, check what backend endpoint handles the discover action. The research mentions `POST /api/discover` exists. Adapt the path accordingly. If no discover endpoint exists at all, create a minimal one in `~/decibel/src/app/api/mobile/discover/route.ts` that:
1. Authenticates via Bearer token
2. Looks up the performer by ID
3. Inserts into `collections` table with `action: "discovered"`
4. Returns `{ success: true, performer: { id, name, slug } }`

Replace the console.log placeholders from Plan 02-02 with real mutation calls:

**onAdd callback:**
```typescript
const addMutation = useAddArtist();
const router = useRouter();

const handleAdd = () => {
  const artist = validateMutation.data?.artist;
  if (!artist) return;

  addMutation.mutate({
    spotifyId: artist.spotify_id || "",
    name: artist.name,
    photoUrl: artist.photo_url,
    genres: artist.genres,
    followers: artist.follower_count || 0,
    monthlyListeners: artist.monthly_listeners ?? undefined,
  }, {
    onSuccess: (result) => {
      // Navigate to artist profile
      router.push(`/artist/${result.performer.slug}`);
      // Reset the paste screen
      setPastedUrl("");
      validateMutation.reset();
    },
  });
};
```

**onDiscover callback:**
```typescript
const discoverMutation = useDiscoverArtist();

const handleDiscover = () => {
  const existing = validateMutation.data?.existing_performer;
  if (!existing) return;

  discoverMutation.mutate({ performerId: existing.id }, {
    onSuccess: (result) => {
      router.push(`/artist/${result.performer.slug}`);
      setPastedUrl("");
      validateMutation.reset();
    },
  });
};
```
