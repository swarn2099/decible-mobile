```typescript
import { createSupabaseAdmin } from './supabase-admin';
import { ArtistSubscription } from './types/artist-dashboard';

export type FeatureAccess = 'full' | 'trial' | 'locked';

export async function getSubscription(performerId: string): Promise<ArtistSubscription | null> {
  const admin = createSupabaseAdmin();
  const { data } = await admin
    .from('artist_subscriptions')
    .select('*')
    .eq('artist_id', performerId)
    .single();
  return data;
}

export function getFeatureAccess(sub: ArtistSubscription | null): FeatureAccess {
  if (!sub) return 'locked';
  if (sub.plan === 'pro' && sub.status === 'active') return 'full';
  if (sub.plan === 'trial' && sub.status === 'trialing') {
    const trialEnd = new Date(sub.trial_ends_at!);
    if (trialEnd > new Date()) return 'trial';
    return 'locked';
  }
  return 'locked';
}

export function isFeatureUnlocked(access: FeatureAccess): boolean {
  return access === 'full' || access === 'trial';
}

export function getTrialDaysLeft(sub: ArtistSubscription | null): number {
  if (!sub?.trial_ends_at) return 0;
  const diff = new Date(sub.trial_ends_at).getTime() - Date.now();
  return Math.max(0, Math.ceil(diff / (1000 * 60 * 60 * 24)));
}
```
