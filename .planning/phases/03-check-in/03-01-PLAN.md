```typescript
import { NextRequest, NextResponse } from "next/server";
import { createClient } from "@supabase/supabase-js";

const admin = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

async function getAuthEmail(req: NextRequest): Promise<string | null> {
  const auth = req.headers.get("authorization");
  if (!auth?.startsWith("Bearer ")) return null;
  const { data, error } = await admin.auth.getUser(auth.slice(7));
  if (error || !data.user?.email) return null;
  return data.user.email;
}
```

First, create the migration script at `~/decibel/scripts/create-user-tagged-events.sql`:

```sql
CREATE TABLE IF NOT EXISTS user_tagged_events (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  fan_id      uuid NOT NULL REFERENCES fans(id) ON DELETE CASCADE,
  venue_id    uuid NOT NULL REFERENCES venues(id) ON DELETE CASCADE,
  performer_id uuid NOT NULL REFERENCES performers(id) ON DELETE CASCADE,
  event_date  date NOT NULL,
  created_at  timestamptz DEFAULT now()
);
CREATE INDEX IF NOT EXISTS user_tagged_events_venue_date_idx ON user_tagged_events(venue_id, event_date);
CREATE UNIQUE INDEX IF NOT EXISTS user_tagged_events_unique_idx ON user_tagged_events(fan_id, venue_id, performer_id, event_date);
```

Run the migration via psql against the Supabase database:
```bash
psql "$DATABASE_URL" -f ~/decibel/scripts/create-user-tagged-events.sql
```

If psql is not available, run the SQL via the Supabase admin client in a one-time Node script using `admin.rpc('exec_sql', { sql: ... })` or by executing the statements through the supabase-js `.from()` raw query pattern.

Then create `~/decibel/src/app/api/mobile/check-in/route.ts`:

POST handler accepting JSON body: `{ venue_id: string, performer_ids: string[], local_date: string }`

Steps:
1. Authenticate via `getAuthEmail(req)` — return 401 if null
2. Look up fan by email: `admin.from('fans').select('id').eq('email', email).single()`
3. Validate required fields (venue_id, performer_ids non-empty array, local_date as YYYY-MM-DD string)
4. Duplicate check: query `collections` WHERE fan_id + venue_id + event_date = local_date. If rows exist, return `{ already_checked_in: true, existing_stamps: [...] }` with status 200 (NOT an error)
5. For each performer_id in the array, insert into `collections`: `{ fan_id, performer_id, venue_id, event_date: local_date, capture_method: 'location', verified: true }`
6. For each performer, upsert `fan_tiers`: increment scan_count (same pattern as /api/collect)
7. Return `{ stamps: [{ performer_id, performer_name, performer_photo }...], venue_name, event_date: local_date, already_checked_in: false }`

Use local_date from the request body directly (CHK-07) — do NOT compute date server-side. This fixes the UTC midnight bug for late-night shows.

If "No live music" is selected on the client, the client simply does not call this endpoint (CHK-04 is frontend-only).

Steps:
1. Authenticate via `getAuthEmail(req)` — return 401 if null
2. Look up fan by email
3. Validate required fields
4. Insert into `user_tagged_events`: `{ fan_id, venue_id, performer_id, event_date: local_date }` — use upsert with onConflict to handle duplicate gracefully (the unique index on fan_id+venue_id+performer_id+event_date prevents duplicates)

7. Query how many user_tagged_events exist for this venue + date to return crowdsourced lineup count
8. Return `{ stamp: { performer_id, performer_name, performer_photo }, venue_name, event_date: local_date, crowdsourced_lineup_count: number }`

The crowdsourced_lineup_count tells the UI how many performers have been tagged at this venue tonight by all users (CHK-09).
